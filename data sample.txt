# 1. Import necessary libraries
import pandas as pd               # For handling data
import matplotlib.pyplot as plt   # For creating charts

# 2. Create a sample dataset (you can replace this with your own data)
data = {
    'Age Group': ['0-20', '21-40', '41-60', '61+'],
    'Population (%)': [34.1, 41.1, 19.4, 5.4]
}

# 3. Convert the dictionary into a Pandas DataFrame
df = pd.DataFrame(data)

# 4. Display the dataset (optional)
print(df)

# 5. Plot a bar chart
plt.bar(df['Age Group'], df['Population (%)'], color=['blue', 'green', 'orange', 'red'])

# 6. Add chart title and axis labels
plt.title("India's Population Distribution by Age (Sample)")
plt.xlabel("Age Group")
plt.ylabel("Population (%)")

# 7. Show grid lines for better readability
plt.grid(axis='y', linestyle='--', alpha=0.7)

# 8. Display the chart
plt.show()
